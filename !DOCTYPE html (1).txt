<script>

let dados = JSON.parse(localStorage.getItem("omega")) || []

function salvar(){
localStorage.setItem("omega",JSON.stringify(dados))
}

function medalha(i){
if(i==0)return"🥇"
if(i==1)return"🥈"
if(i==2)return"🥉"
return""
}

function add(){

let t=document.getElementById("entrada").value.split(/\s+/)

if(t.length<4){
alert("Use: Nome ID Chamados Avaliações")
return
}

dados.push({
nome:t[0],
id:t[1],
chamados:Number(t[2]),
avaliacoes:Number(t[3])
})

document.getElementById("entrada").value=""

ordenar()

}

function ordenar(){
dados.sort((a,b)=>b.chamados-a.chamados)
salvar()
render()
}

function atualizarCards(){

document.getElementById("membros").innerText=dados.length

let totalChamados=dados.reduce((s,p)=>s+p.chamados,0)

let totalAval=dados.reduce((s,p)=>s+p.avaliacoes,0)

document.getElementById("totalChamados").innerText=totalChamados
document.getElementById("totalAval").innerText=totalAval

}

function editarCampo(i,campo,valor){

if(campo=="chamados" || campo=="avaliacoes"){
valor=Number(valor)
if(isNaN(valor)) return
}

dados[i][campo]=valor

ordenar()

}

function excluir(i){

if(confirm("Excluir membro?")){

dados.splice(i,1)

salvar()

render()

}

}

function render(lista=dados){

let tb=document.getElementById("tabela")

tb.innerHTML=""

lista.forEach((p,i)=>{

let tr=document.createElement("tr")

if(i==0)tr.className="top1"
if(i==1)tr.className="top2"
if(i==2)tr.className="top3"

tr.innerHTML=`

<td>${medalha(i)}</td>

<td contenteditable="true" onblur="editarCampo(${i},'nome',this.innerText)">
${p.nome}
</td>

<td contenteditable="true" onblur="editarCampo(${i},'id',this.innerText)">
${p.id}
</td>

<td contenteditable="true" onblur="editarCampo(${i},'chamados',this.innerText)">
${p.chamados}
</td>

<td contenteditable="true" onblur="editarCampo(${i},'avaliacoes',this.innerText)">
${p.avaliacoes}
</td>

<td>
<button onclick="excluir(${i})">🗑</button>
</td>

`

tb.appendChild(tr)

})

atualizarCards()

grafico()

}

function buscar(){

let t=document.getElementById("buscar").value.toLowerCase()

let f=dados.filter(p=>
p.nome.toLowerCase().includes(t) ||
p.id.includes(t)
)

render(f)

}

function limpar(){

if(confirm("Apagar tudo?")){
dados=[]
salvar()
render()
}

}

function excel(){

let ws=XLSX.utils.json_to_sheet(dados)

let wb=XLSX.utils.book_new()

XLSX.utils.book_append_sheet(wb,ws,"Omega")

XLSX.writeFile(wb,"Equipe_Omega.xlsx")

}

function pdf(){
window.print()
}

let chart

function grafico(){

let nomes=dados.map(p=>p.nome)
let cham=dados.map(p=>p.chamados)

if(chart)chart.destroy()

chart=new Chart(document.getElementById("grafico"),{

type:"bar",

data:{
labels:nomes,
datasets:[{
label:"Chamados por Analista",
data:cham,
backgroundColor:[
"#0284c7",
"#0ea5e9",
"#38bdf8",
"#7dd3fc",
"#bae6fd"
]
}]
},

options:{
scales:{y:{beginAtZero:true}}
}

})

}

ordenar()
